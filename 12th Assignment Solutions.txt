1. Write a recursive function to print first N natural numbers.

#include<stdio.h>
void Nnatural(int);
int main()
{
    int a;
    printf("Enter a number");
    scanf("%d",&a);
    Nnatural(a);
    return 0;
}
void Nnatural(int n)
{
  int i;
  if(n>0)
  {
     Nnatural(n-1);
     printf("%d ",n);
  }
}
