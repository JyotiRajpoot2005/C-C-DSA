              break;

        case 2:
              printf("Enter lengths of three sides of triangle");

              printf("Enter lengths of three sides of triangle");

6. Program to check whether a year is a leap year or not. Using switch statement.

#include<stdio.h>
int main()
{
    int a,year;
    printf("Enter a number");
    scanf("%d",&a);
    switch(a)
        {
        case 1:
              printf("\nEnter your year");
              scanf("%d",&year);
              if(year%100==0)
              {
                  if(year%400==0)
                    printf("leap year");
                  else
                    printf("not a leap year");
              }
              else
              {
                  if(year%4==0)
                     printf("leap year");
                  else
                    printf("not a leap year");
              }
               break;

        default:
            printf("Invalid number");
        }

    return 0;
}
